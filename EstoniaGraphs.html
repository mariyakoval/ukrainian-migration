# Function to create a clean proportion table
make_prop_table <- function(variable, na.rm = FALSE) {
  if (na.rm) {
    variable <- variable[!is.na(variable)]
  }
  
  tab <- table(variable)
  
  tibble::tibble(
    Category = names(tab),
    Count = as.numeric(tab),
    Percent = round(100 * prop.table(tab), 2)
  )
}
